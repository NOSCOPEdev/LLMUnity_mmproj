111 public static string FromGGUF(string path)
112 {
113 GGUFReader reader = new GGUFReader(path);
114 string name;
115
116 name = FromTemplate(reader.GetStringField("tokenizer.chat_template"));
117 if (name != null) return name;
118
119 name = FromName(reader.GetStringField("general.name"));
120 if (name != null) return name;
121
122 name = FromName(Path.GetFileNameWithoutExtension(path));
123 if (name != null) return name;
124
125 Debug.Log("No chat template could be matched, fallback to ChatML");
126 return DefaultTemplate;

164 public virtual string ComputePrompt(List<ChatMessage> messages, string AIName)
165 {
166 string chatPrompt = PromptPrefix();
167 string systemPrompt = "";
168 int start = 0;
169 if (messages[0].role == "system")
170 {
171 systemPrompt = SystemPrefix() + messages[0].content + SystemSuffix();
172 start = 1;
173 }
174 for (int i = start; i < messages.Count; i += 2)
175 {
176 chatPrompt += RequestPrefix();
177 if (i == 1 && systemPrompt != "") chatPrompt += systemPrompt;
178 chatPrompt += PlayerPrefix(messages[i].role) + PrefixMessageSeparator() + messages[i].content + RequestSuffix();
179 if (i < messages.Count - 1)
180 {
181 chatPrompt += AIPrefix(messages[i + 1].role) + PrefixMessageSeparator() + messages[i + 1].content + PairSuffix();
182 }
183 }
184 chatPrompt += AIPrefix(AIName);
185 return chatPrompt;
186 }
