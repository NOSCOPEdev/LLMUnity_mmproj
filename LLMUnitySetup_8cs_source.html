189 public static async Task DownloadFile(
190 string fileUrl, string savePath, bool overwrite = false,
191 TaskCallback<string> callback = null, Callback<float> progresscallback = null,
192 bool async = true
193 )
194 {
195 // download a file to the specified path
196 if (File.Exists(savePath) && !overwrite)
197 {
198 Log($"File already exists at: {savePath}");
199 }
200 else
201 {
202 Log($"Downloading {fileUrl}...");
203 string tmpPath = Path.Combine(Application.temporaryCachePath, Path.GetFileName(savePath));
204
205 WebClient client = new WebClient();
206 DownloadStatus downloadStatus = new DownloadStatus(progresscallback);
207 client.DownloadProgressChanged += downloadStatus.DownloadProgressChanged;
208 if (async)
209 {
210 await client.DownloadFileTaskAsync(fileUrl, tmpPath);
211 }
212 else
213 {
214 client.DownloadFile(fileUrl, tmpPath);
215 }
216
217 AssetDatabase.StartAssetEditing();
218 Directory.CreateDirectory(Path.GetDirectoryName(savePath));
219 File.Move(tmpPath, savePath);
220 AssetDatabase.StopAssetEditing();
221 Log($"Download complete!");
222 }
223
224 progresscallback?.Invoke(1f);
225 if (callback != null) await callback.Invoke(savePath);
226 }

228 public static async Task<string> AddAsset(string assetPath, string basePath)
229 {
230 if (!File.Exists(assetPath))
231 {
232 LogError($"{assetPath} does not exist!");
233 return null;
234 }
235 // add an asset to the basePath directory if it is not already there and return the relative path
236 string basePathSlash = basePath.Replace('\\', '/');
237 string fullPath = Path.GetFullPath(assetPath).Replace('\\', '/');
238 Directory.CreateDirectory(basePathSlash);
239 if (!fullPath.StartsWith(basePathSlash))
240 {
241 // if the asset is not in the assets dir copy it over
242 fullPath = Path.Combine(basePathSlash, Path.GetFileName(assetPath));
243 Log($"copying {assetPath} to {fullPath}");
244 AssetDatabase.StartAssetEditing();
245 await Task.Run(() =>
246 {
247 foreach (string filename in new string[] {fullPath, fullPath + ".meta"})
248 {
249 if (File.Exists(fullPath))
250 File.Delete(fullPath);
251 }
252 File.Copy(assetPath, fullPath);
253 });
254 AssetDatabase.StopAssetEditing();
255 Log("copying complete!");
256 }
257 return fullPath.Substring(basePathSlash.Length + 1);
258 }
