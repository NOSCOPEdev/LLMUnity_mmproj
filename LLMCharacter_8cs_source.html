406 public async Task<string> Chat(string query, Callback<string> callback = null, EmptyCallback completionCallback = null, bool addToHistory = true)
407 {
408 // handle a chat message by the user
409 // call the callback function while the answer is received
410 // call the completionCallback function when the answer is fully received
411 await LoadTemplate();
412 await InitNKeep();
413
414 string json;
415 await chatLock.WaitAsync();
416 try
417 {
418 AddPlayerMessage(query);
419 string prompt = template.ComputePrompt(chat, AIName);
420 json = JsonUtility.ToJson(GenerateRequest(prompt));
421 chat.RemoveAt(chat.Count - 1);
422 }
423 finally
424 {
425 chatLock.Release();
426 }
427
428 string result = await CompletionRequest(json, callback);
429
430 if (addToHistory && result != null)
431 {
432 await chatLock.WaitAsync();
433 try
434 {
435 AddPlayerMessage(query);
436 AddAIMessage(result);
437 }
438 finally
439 {
440 chatLock.Release();
441 }
442 if (save != "") _ = Save(save);
443 }
444
445 completionCallback?.Invoke();
446 return result;
447 }

575 protected Ret ConvertContent<Res, Ret>(string response, ContentCallback<Res, Ret> getContent = null)
576 {
577 // template function to convert the json received and get the content
578 if (response == null) return default;
579 response = response.Trim();
580 if (response.StartsWith("data: "))
581 {
582 string responseArray = "";
583 foreach (string responsePart in response.Replace("\n\n", "").Split("data: "))
584 {
585 if (responsePart == "") continue;
586 if (responseArray != "") responseArray += ",\n";
587 responseArray += responsePart;
588 }
589 response = $"{{\"data\": [{responseArray}]}}";
590 }
591 return getContent(JsonUtility.FromJson<Res>(response));
592 }

618 protected async Task<Ret> PostRequestLocal<Res, Ret>(string json, string endpoint, ContentCallback<Res, Ret> getContent, Callback<Ret> callback = null)
619 {
620 // send a post request to the server and call the relevant callbacks to convert the received content and handle it
621 // this function has streaming functionality i.e. handles the answer while it is being received
622 string callResult = null;
623 bool callbackCalled = false;
624 while (!llm.failed && !llm.started) await Task.Yield();
625 switch (endpoint)
626 {
627 case "tokenize":
628 callResult = await llm.Tokenize(json);
629 break;
630 case "detokenize":
631 callResult = await llm.Detokenize(json);
632 break;
633 case "slots":
634 callResult = await llm.Slot(json);
635 break;
636 case "completion":
637 Callback<string> callbackString = null;
638 if (stream && callback != null)
639 {
640 if (typeof(Ret) == typeof(string))
641 {
642 callbackString = (strArg) =>
643 {
644 callback(ConvertContent(strArg, getContent));
645 };
646 }
647 else
648 {
649 Debug.LogError($"wrong callback type, should be string");
650 }
651 callbackCalled = true;
652 }
653 callResult = await llm.Completion(json, callbackString);
654 break;
655 default:
656 Debug.LogError($"Unknown endpoint {endpoint}");
657 break;
658 }
659
660 Ret result = ConvertContent(callResult, getContent);
661 if (!callbackCalled) callback?.Invoke(result);
662 return result;
663 }

665 protected async Task<Ret> PostRequestRemote<Res, Ret>(string json, string endpoint, ContentCallback<Res, Ret> getContent, Callback<Ret> callback = null)
666 {
667 // send a post request to the server and call the relevant callbacks to convert the received content and handle it
668 // this function has streaming functionality i.e. handles the answer while it is being received
669 if (endpoint == "slots")
670 {
671 Debug.LogError("Saving and loading is not currently supported in remote setting");
672 return default;
673 }
674
675 Ret result = default;
676 byte[] jsonToSend = new System.Text.UTF8Encoding().GetBytes(json);
677 using (var request = UnityWebRequest.Put($"{host}:{port}/{endpoint}", jsonToSend))
678 {
679 WIPRequests.Add(request);
680
681 request.method = "POST";
682 if (requestHeaders != null)
683 {
684 for (int i = 0; i < requestHeaders.Count; i++)
685 request.SetRequestHeader(requestHeaders[i].Item1, requestHeaders[i].Item2);
686 }
687
688 // Start the request asynchronously
689 var asyncOperation = request.SendWebRequest();
690 float lastProgress = 0f;
691 // Continue updating progress until the request is completed
692 while (!asyncOperation.isDone)
693 {
694 float currentProgress = request.downloadProgress;
695 // Check if progress has changed
696 if (currentProgress != lastProgress && callback != null)
697 {
698 callback?.Invoke(ConvertContent(request.downloadHandler.text, getContent));
699 lastProgress = currentProgress;
700 }
701 // Wait for the next frame
702 await Task.Yield();
703 }
704 WIPRequests.Remove(request);
705 if (request.result != UnityWebRequest.Result.Success) Debug.LogError(request.error);
706 else result = ConvertContent(request.downloadHandler.text, getContent);
707 callback?.Invoke(result);
708 }
709 return result;
710 }
