322 private async Task StartLLMServer()
323 {
324 bool portInUse = asynchronousStartup ? await IsServerReachableAsync() : IsServerReachable();
325 if (portInUse)
326 {
327 Debug.LogError($"Port {port} is already in use, please use another port or kill all llamafile processes using it!");
328 return;
329 }
330
331 // Start the LLM server in a cross-platform way
332 if (model == "")
333 {
334 Debug.LogError("No model file provided!");
335 return;
336 }
337 string modelPath = LLMUnitySetup.GetAssetPath(model);
338 if (!File.Exists(modelPath))
339 {
340 Debug.LogError($"File {modelPath} not found!");
341 return;
342 }
343
344 string loraPath = "";
345 if (lora != "")
346 {
347 loraPath = LLMUnitySetup.GetAssetPath(lora);
348 if (!File.Exists(loraPath))
349 {
350 Debug.LogError($"File {loraPath} not found!");
351 return;
352 }
353 }
354
355 int slots = parallelPrompts == -1 ? GetListeningClients().Count + 1 : parallelPrompts;
356 string arguments = $" --port {port} -m {EscapeSpaces(modelPath)} -c {contextSize} -b {batchSize} --log-disable --nobrowser -np {slots}";
357 if (remote) arguments += $" --host 0.0.0.0";
358 if (numThreads > 0) arguments += $" -t {numThreads}";
359 if (loraPath != "") arguments += $" --lora {EscapeSpaces(loraPath)}";
360
361 string noGPUArgument = " --nocompile --gpu disable";
362 string GPUArgument = numGPULayers <= 0 ? noGPUArgument : $" -ngl {numGPULayers}";
363 LLMUnitySetup.makeExecutable(server);
364
365 RunServerCommand(server, arguments + GPUArgument);
366 if (asynchronousStartup) await WaitOneASync(serverBlock, TimeSpan.FromSeconds(60));
367 else serverBlock.WaitOne(60000);
368
369 if (process.HasExited && mmapCrash)
370 {
371 Debug.Log("Mmap error, fallback to no mmap use");
372 serverBlock.Reset();
373 arguments += " --no-mmap";
374
375 RunServerCommand(server, arguments + GPUArgument);
376 if (asynchronousStartup) await WaitOneASync(serverBlock, TimeSpan.FromSeconds(60));
377 else serverBlock.WaitOne(60000);
378 }
379
380 if (process.HasExited && numGPULayers > 0)
381 {
382 Debug.Log("GPU failed, fallback to CPU");
383 serverBlock.Reset();
384
385 RunServerCommand(server, arguments + noGPUArgument);
386 if (asynchronousStartup) await WaitOneASync(serverBlock, TimeSpan.FromSeconds(60));
387 else serverBlock.WaitOne(60000);
388 }
389
390 if (process.HasExited) Debug.LogError("Server could not be started!");
391 else LLMUnitySetup.SaveServerPID(process.Id);
392 }

422 private static Task WaitOneASync(WaitHandle handle, TimeSpan timeout)
423 {
424 var tcs = new TaskCompletionSource<object>();
425 var registration = ThreadPool.RegisterWaitForSingleObject(handle, (state, timedOut) =>
426 {
427 var localTcs = (TaskCompletionSource<object>)state;
428 if (timedOut)
429 localTcs.TrySetCanceled();
430 else
431 localTcs.TrySetResult(null);
432 }, tcs, timeout, executeOnlyOnce: true);
433 tcs.Task.ContinueWith((_, state) => ((RegisteredWaitHandle)state).Unregister(null), registration, TaskScheduler.Default);
434 return tcs.Task;
435 }
