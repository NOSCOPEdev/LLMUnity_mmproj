172 protected virtual string GetLlamaccpArguments()
173 {
174 // Start the LLM server in a cross-platform way
175 if (model == "")
176 {
177 Debug.LogError("No model file provided!");
178 return null;
179 }
180 string modelPath = LLMUnitySetup.GetAssetPath(model);
181 if (!File.Exists(modelPath))
182 {
183 Debug.LogError($"File {modelPath} not found!");
184 return null;
185 }
186 string loraPath = "";
187 if (lora != "")
188 {
189 loraPath = LLMUnitySetup.GetAssetPath(lora);
190 if (!File.Exists(loraPath))
191 {
192 Debug.LogError($"File {loraPath} not found!");
193 return null;
194 }
195 }
196
197 int slots = GetNumClients();
198 string arguments = $"-m \"{modelPath}\" -c {contextSize} -b {batchSize} --log-disable -np {slots}";
199 if (remote) arguments += $" --port {port} --host 0.0.0.0";
200 if (numThreads > 0) arguments += $" -t {numThreads}";
201 if (loraPath != "") arguments += $" --lora \"{loraPath}\"";
202 arguments += $" --slot-save-path \"{slotSaveDir}\"";
203 arguments += $" -ngl {numGPULayers}";
204 return arguments;
205 }

233 private void StartLLMServer()
234 {
235 started = false;
236 failed = false;
237 string arguments = GetLlamaccpArguments();
238 if (arguments == null) return;
239 bool useGPU = numGPULayers > 0;
240 Debug.Log($"Server command: {arguments}");
241
242 foreach (string arch in LLMLib.PossibleArchitectures(useGPU))
243 {
244 string error;
245 try
246 {
247 InitLib(arch);
248 InitServer(arguments);
249 Debug.Log($"Using architecture: {arch}");
250 break;
251 }
252 catch (LLMException e)
253 {
254 error = e.Message;
255 Destroy();
256 }
257 catch (Exception e)
258 {
259 error = $"{e.GetType()}: {e.Message}";
260 }
261 Debug.Log($"Tried architecture: {arch}, " + error);
262 }
263 if (llmlib == null)
264 {
265 Debug.LogError("LLM service couldn't be created");
266 failed = true;
267 return;
268 }
269 StartService();
270 Debug.Log("LLM service created");
271 }

351 void CheckLLMStatus(bool log = true)
352 {
353 if (llmlib == null) {return;}
354 IntPtr stringWrapper = llmlib.StringWrapper_Construct();
355 int status = llmlib.LLM_Status(LLMObject, stringWrapper);
356 string result = llmlib.GetStringWrapperResult(stringWrapper);
357 llmlib.StringWrapper_Delete(stringWrapper);
358 string message = $"LLM {status}: {result}";
359 if (status > 0)
360 {
361 if (log) Debug.LogError(message);
362 throw new LLMException(message, status);
363 }
364 else if (status < 0)
365 {
366 if (log) Debug.LogWarning(message);
367 }
