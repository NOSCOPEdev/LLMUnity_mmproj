218 protected virtual string GetLlamaccpArguments()
219 {
220 // Start the LLM server in a cross-platform way
221 if (model == "")
222 {
223 LLMUnitySetup.LogError("No model file provided!");
224 return null;
225 }
226 string modelPath = GetModelLoraPath(model);
227 if (!File.Exists(modelPath))
228 {
229 LLMUnitySetup.LogError($"File {modelPath} not found!");
230 return null;
231 }
232 string loraPath = "";
233 if (lora != "")
234 {
235 loraPath = GetModelLoraPath(lora);
236 if (!File.Exists(loraPath))
237 {
238 LLMUnitySetup.LogError($"File {loraPath} not found!");
239 return null;
240 }
241 }
242
243 int numThreadsToUse = numThreads;
244 if (Application.platform == RuntimePlatform.Android && numThreads <= 0) numThreadsToUse = LLMUnitySetup.AndroidGetNumBigCores();
245
246 int slots = GetNumClients();
247 string arguments = $"-m \"{modelPath}\" -c {contextSize} -b {batchSize} --log-disable -np {slots}";
248 if (remote) arguments += $" --port {port} --host 0.0.0.0";
249 if (numThreadsToUse > 0) arguments += $" -t {numThreadsToUse}";
250 if (loraPath != "") arguments += $" --lora \"{loraPath}\"";
251 arguments += $" -ngl {numGPULayers}";
252 return arguments;
253 }

268 private void StartLLMServer(string arguments)
269 {
270 started = false;
271 failed = false;
272 bool useGPU = numGPULayers > 0;
273 LLMUnitySetup.Log($"Server command: {arguments}");
274
275 foreach (string arch in LLMLib.PossibleArchitectures(useGPU))
276 {
277 string error;
278 try
279 {
280 InitLib(arch);
281 InitServer(arguments);
282 LLMUnitySetup.Log($"Using architecture: {arch}");
283 break;
284 }
285 catch (LLMException e)
286 {
287 error = e.Message;
288 Destroy();
289 }
290 catch (Exception e)
291 {
292 error = $"{e.GetType()}: {e.Message}";
293 }
294 LLMUnitySetup.Log($"Tried architecture: {arch}, " + error);
295 }
296 if (llmlib == null)
297 {
298 LLMUnitySetup.LogError("LLM service couldn't be created");
299 failed = true;
300 return;
301 }
302 StartService();
303 LLMUnitySetup.Log("LLM service created");
304 }

384 void CheckLLMStatus(bool log = true)
385 {
386 if (llmlib == null) {return;}
387 IntPtr stringWrapper = llmlib.StringWrapper_Construct();
388 int status = llmlib.LLM_Status(LLMObject, stringWrapper);
389 string result = llmlib.GetStringWrapperResult(stringWrapper);
390 llmlib.StringWrapper_Delete(stringWrapper);
391 string message = $"LLM {status}: {result}";
392 if (status > 0)
393 {
394 if (log) LLMUnitySetup.LogError(message);
395 throw new LLMException(message, status);
396 }
397 else if (status < 0)
398 {
399 if (log) LLMUnitySetup.LogWarning(message);
400 }
