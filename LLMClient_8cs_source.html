446 public async Task<string> Chat(string query, Callback<string> callback = null, EmptyCallback completionCallback = null, bool addToHistory = true)
447 {
448 // handle a chat message by the user
449 // call the callback function while the answer is received
450 // call the completionCallback function when the answer is fully received
451 await InitNKeep();
452
453 string json;
454 lock (chatPromptLock) {
455 AddPlayerMessage(query);
456 string prompt = template.ComputePrompt(chat, AIName);
457 json = JsonUtility.ToJson(GenerateRequest(prompt));
458 chat.RemoveAt(chat.Count - 1);
459 }
460
461 string result = await CompletionRequest(json, callback);
462
463 if (addToHistory && result != null)
464 {
465 lock (chatAddLock) {
466 AddPlayerMessage(query);
467 AddAIMessage(result);
468 }
469 }
470
471 completionCallback?.Invoke();
472 return result;
473 }

643 async Task<Ret> PostRequest<Res, Ret>(string json, string endpoint, ContentCallback<Res, Ret> getContent, Callback<Ret> callback = null)
644 {
645 // send a post request to the server and call the relevant callbacks to convert the received content and handle it
646 // this function has streaming functionality i.e. handles the answer while it is being received
647 Ret result = default;
648 string errorMessage = "";
649 if (host == "localhost" && server == null) errorMessage += "No server found!";
650 if (server != null && !server.serverListening) errorMessage += "Server is not listening!";
651 if (server != null && LLMUnitySetup.NumServersForPort(port) > 1) errorMessage += "Multiple servers found for port!";
652 if (errorMessage != "")
653 {
654 Debug.LogError(errorMessage);
655 return result;
656 }
657
658 byte[] jsonToSend = new System.Text.UTF8Encoding().GetBytes(json);
659 using (var request = UnityWebRequest.Put($"{host}:{port}/{endpoint}", jsonToSend))
660 {
661 WIPRequests.Add(request);
662
663 request.method = "POST";
664 if (requestHeaders != null)
665 {
666 for (int i = 0; i < requestHeaders.Count; i++)
667 request.SetRequestHeader(requestHeaders[i].Item1, requestHeaders[i].Item2);
668 }
669
670 // Start the request asynchronously
671 var asyncOperation = request.SendWebRequest();
672 float lastProgress = 0f;
673 // Continue updating progress until the request is completed
674 while (!asyncOperation.isDone)
675 {
676 float currentProgress = request.downloadProgress;
677 // Check if progress has changed
678 if (currentProgress != lastProgress && callback != null)
679 {
680 callback?.Invoke(ConvertContent(request.downloadHandler.text, getContent));
681 lastProgress = currentProgress;
682 }
683 // Wait for the next frame
684 await Task.Yield();
685 }
686 WIPRequests.Remove(request);
687 if (request.result != UnityWebRequest.Result.Success) Debug.LogError(request.error);
688 else result = ConvertContent(request.downloadHandler.text, getContent);
689 callback?.Invoke(result);
690 }
691 return result;
692 }
